>>> str2='<user01@py.com><user02@yp.com><user03@pp.edu>'
>>> re.findall(r'(\w+@[a-z]{2}\.[a-z]{3})',str2)
['user01@py.com', 'user02@yp.com', 'ruse03@pp.edu']
>>> re.sub(r'user','test',str2)
'<test01@py.com><test02@yp.com><test03@pp.edu>'
#parameter 'count' define how many replacing to perform
#default 0,which mean no limit
>>> re.sub(r'test','user',str2,count=2)
'<user01@py.com><user02@yp.com><test03@pp.edu>'
>>> str3='Hello, Nice to meet you'
>>> re.split(r'\s',str3)
['Hello,', 'Nice', 'to', 'meet', 'you']
>>> str3
'Hello, Nice to meet you'
#compilation could make matching faster
>>> c=re.compile(r'\w*e\w*')
>>> c.findall(str3)
['Hello', 'Nice', 'meet']
>>> c.sub(lambda m:'<'+m.group(0)+'>',str3)
